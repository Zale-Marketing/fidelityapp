**New computeRecipientCount() async function** (call on mount and whenever selectedProgram or selectedTag changes via useEffect):
```typescript
async function computeRecipientCount() {
  if (!merchantId) return
  setCountLoading(true)

  // Determine which program IDs to include
  const programIds = selectedProgram === 'all'
    ? programs.map(p => p.id)
    : [selectedProgram]

  if (programIds.length === 0) { setRecipientCount(0); setCountLoading(false); return }

  if (selectedTag === 'all') {
    // Count distinct card_holder_ids on active cards in target programs
    // Use card_holder_id IS NOT NULL to count only cards with known customers
    const { data: cards } = await supabase
      .from('cards')
      .select('card_holder_id')
      .in('program_id', programIds)
      .eq('status', 'active')
      .not('card_holder_id', 'is', null)
    const uniqueHolders = new Set((cards || []).map(c => c.card_holder_id))
    setRecipientCount(uniqueHolders.size)
  } else {
    // Get card_holder_ids that have the selected tag
    const { data: holderTags } = await supabase
      .from('card_holder_tags')
      .select('card_holder_id')
      .eq('tag_id', selectedTag)

    const taggedHolderIds = (holderTags || []).map(ht => ht.card_holder_id)

    if (taggedHolderIds.length === 0) { setRecipientCount(0); setCountLoading(false); return }

    // Count cards for tagged holders in target programs
    const { data: cards } = await supabase
      .from('cards')
      .select('card_holder_id')
      .in('program_id', programIds)
      .in('card_holder_id', taggedHolderIds)
      .eq('status', 'active')

    const uniqueHolders = new Set((cards || []).map(c => c.card_holder_id))
    setRecipientCount(uniqueHolders.size)
  }

  setCountLoading(false)
}
```

Add useEffect to recompute count when filters change (after merchantId is set and programs loaded):
```typescript
useEffect(() => {
  if (merchantId && programs.length >= 0) {
    computeRecipientCount()
  }
}, [merchantId, selectedProgram, selectedTag, programs])
```

Note: use 300ms debounce via useRef timeout to avoid firing multiple simultaneous queries when filters change rapidly. Pattern: `const countTimer = useRef<ReturnType<typeof setTimeout> | null>(null)` — clear and reset on each effect trigger.
